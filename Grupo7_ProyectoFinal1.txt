    let completedGames = [];
    let audioContext;

    let introAudioMemory = new Audio('Juego de memorias  .ogg');

    function initApp() {
      loadProgress();
      updateMenu();
      initAudio();
      document.addEventListener('keydown', e => { if (e.key === 'Escape') closeModal(); });
    }

    function initAudio() {
      audioContext = new (window.AudioContext || window.webkitAudioContext)();
    }

    function loadProgress() {
      completedGames = JSON.parse(localStorage.getItem('babyLogicCompleted') || '[]');
    }
    function saveProgress() {
      localStorage.setItem('babyLogicCompleted', JSON.stringify(completedGames));
    }
    function markCompleted(num) {
      if (!completedGames.includes(num)) {
        completedGames.push(num);
        saveProgress();
        updateMenu();
      }
    }
    function updateMenu() {
      document.querySelectorAll('.game-card').forEach(card => {
        const num = parseInt(card.dataset.game);
        card.classList.toggle('completed', completedGames.includes(num));
      });
      const percent = Math.round((completedGames.length / 14) * 100);
      document.getElementById('progress-bar').style.width = percent + '%';
      document.getElementById('progress-text').textContent = `${completedGames.length}/14`;
    }

    function playTone(freq, duration, type = 'triangle', vol = 0.4) {
      if (!audioContext) initAudio();
      const osc = audioContext.createOscillator();
      const gain = audioContext.createGain();
      osc.type = type;
      osc.frequency.value = freq;
      gain.gain.value = vol;
      osc.connect(gain).connect(audioContext.destination);
      osc.start();
      setTimeout(() => osc.stop(), duration);
    }
    function playSuccessSound() {
      playTone(700, 120); setTimeout(() => playTone(1100, 120), 80);
      setTimeout(() => playTone(1400, 280), 200);
    }
    function playErrorSound() {
      playTone(320, 420, 'sawtooth', 0.5);
    }

    function toggleMusic() {
      musicPlaying = !musicPlaying;
      const btn = document.getElementById('music-toggle');
      if (musicPlaying) {
        btn.innerHTML = `⏹️ <span class="text-sm uppercase font-black">Pause Music</span>`;
        startMusicLoop();
      } else {
        btn.innerHTML = `🎧 <span class="text-sm uppercase font-black">Audio System</span>`;
        clearInterval(musicTimer);
      }
    }
    function startMusicLoop() {
      if (musicTimer) clearInterval(musicTimer);
      let index = 0;
      musicTimer = setInterval(() => {
        if (!musicPlaying) { clearInterval(musicTimer); return; }
        playTone(melodyNotes[index], 280, 'sine', 0.15);
        index = (index + 1) % melodyNotes.length;
      }, 340);
    }

    function launchConfetti() {
      const emojis = ['⚡','💎','✨','🚀','🔥','💻'];
      for (let i = 0; i < 50; i++) {

        c.textContent = emojis[Math.floor(Math.random()*emojis.length)];
        c.style.position = 'fixed';
        c.style.left = Math.random() * 100 + 'vw';
        c.style.top = '-50px';
        c.style.fontSize = '24px';
        c.style.zIndex = 9999;
        document.body.appendChild(c);
        const duration = 2000 + Math.random() * 2000;
        c.animate([
          { transform: 'translateY(0) rotate(0deg)', opacity: 1 },
          { transform: `translateY(${window.innerHeight + 100}px) rotate(360deg)`, opacity: 0 }
        ], { duration: duration });
        setTimeout(() => c.remove(), duration);
      }
    }

    function openGame(num) {

    function closeModal() {
      [introAudioMemory, introAudioColors, introAudioFruits, introAudioVegetables, 
       introAudioClassify, introAudioPuzzle, introAudioSize, introAudioShapes, 
       introAudioPatterns, introAudioShadow, introAudioCount, introAudioMaze,

        a.pause();
        a.currentTime = 0;
      });
      document.getElementById('modal').classList.add('hidden');
    }

    function initMemoryGame() {
      let cards = [...['🐶','🐱','🐭','🐹','🐰','🦊','🐻','🐼'], ...['🐶','🐱','🐭','🐹','🐰','🦊','🐻','🐼']].sort(() => Math.random() - 0.5);
      let flipped = []; let matched = []; let moves = 0;
      const grid = document.getElementById('memory-grid');
      grid.innerHTML = '';
      cards.forEach((emoji, i) => {
        const card = document.createElement('div');
        card.className = 'w-16 h-16 bg-white/5 border border-white/10 rounded-2xl flex items-center justify-center text-4xl cursor-pointer hover:bg-white/10 transition-all';
        card.dataset.emoji = emoji;
        card.onclick = () => {
          if (flipped.length >= 2 || card.classList.contains('flipped') || card.classList.contains('matched')) return;
          card.classList.add('flipped');
          card.textContent = card.dataset.emoji;
          card.style.borderColor = "#38bdf8";
          flipped.push(card);
          moves++;
          document.getElementById('moves').textContent = moves;
          if (flipped.length === 2) {
            if (flipped[0].dataset.emoji === flipped[1].dataset.emoji) {
              flipped.forEach(c => { c.classList.add('matched'); c.style.backgroundColor = "rgba(16, 185, 129, 0.2)"; });
              matched.push(...flipped);
              playSuccessSound();
              flipped = [];
              if (matched.length === cards.length) setTimeout(() => showWinScreen(1), 400);
            } else {
              setTimeout(() => {
                flipped.forEach(c => { c.classList.remove('flipped'); c.textContent = ''; c.style.borderColor = "rgba(255,255,255,0.1)"; });
                flipped = [];
                playErrorSound();
              }, 800);
            }
          }
        };
        grid.appendChild(card);
      });
      introAudioMemory.play().catch(e => console.log("Audio intro 1"));
    }

    function initColorGame(content) {
      const colors = [{emoji:'🔴',name:'rojo',class:'bg-red-500'},{emoji:'🔵',name:'azul',class:'bg-blue-500'},{emoji:'🟡',name:'amarillo',class:'bg-yellow-400'},{emoji:'🟢',name:'verde',class:'bg-green-500'}];
      let score = 0, round = 0;
      content.innerHTML = `<h2 class="text-2xl font-bold text-center mb-6 uppercase tracking-widest text-blue-400">Filtro Cromático</h2><p id="color-q" class="text-xl text-center mb-10">Selecciona el color</p><div id="color-options" class="grid grid-cols-2 gap-4"></div><p class="text-center mt-8 font-mono text-emerald-400" id="color-score">SYST_STATUS: 0 / 5</p>`;
      
      introAudioColors.play().catch(e => console.log("Audio intro 2"));

      const nextRound = () => {
        if (round >= 5) { showWinScreen(2, `PRECISIÓN: ${score*20}%`); return; }
        round++;
        const correct = colors[Math.floor(Math.random() * colors.length)];
        const optsDiv = document.getElementById('color-options');
        optsDiv.innerHTML = '';
        colors.slice().sort(() => Math.random()-0.5).forEach(opt => {
          const btn = document.createElement('button');
          btn.className = `h-24 rounded-2xl text-white text-5xl shadow-lg border border-white/20 ${opt.class} hover:scale-105 active:scale-95 transition-all`;
          btn.textContent = opt.emoji;
          btn.onclick = () => {
            if (opt.name === correct.name) { score++; playSuccessSound(); nextRound(); }
            else { playErrorSound(); }
            document.getElementById('color-score').textContent = `SYST_STATUS: ${score} / 5`;
          };
          optsDiv.appendChild(btn);
        });
      }
      nextRound();
    }

    function initFruitGame(content, gameNum) {
      const isFruit = gameNum === 3;
      const items = isFruit ? [{e:'🍎',n:'manzana'},{e:'🍌',n:'plátano'},{e:'🍓',n:'fresa'},{e:'🍊',n:'naranja'}] : [{e:'🥕',n:'zanahoria'},{e:'🥦',n:'brócoli'},{e:'🌽',n:'maíz'},{e:'🍅',n:'tomate'}];
      let score = 0, round = 0;
      content.innerHTML = `<h2 class="text-2xl font-bold text-center mb-6 uppercase tracking-widest text-emerald-400">Scanner Orgánico</h2><p id="question" class="text-xl text-center mb-8">Escucha el audio y selecciona el elemento correcto</p><div id="options" class="grid grid-cols-2 gap-6 text-7xl"></div><p class="text-center mt-8 font-mono text-emerald-400" id="fscore">0 / 5</p>`;
      if (isFruit) introAudioFruits.play(); else introAudioVegetables.play(); 
      const next = () => {
        if (round >= 5) { showWinScreen(gameNum); return; }
        round++;
        const correct = items[Math.floor(Math.random()*items.length)];
        const optsDiv = document.getElementById('options');
        optsDiv.innerHTML = '';
        items.slice().sort(() => Math.random()-0.5).forEach(item => {
          const div = document.createElement('div');
          div.className = `bg-white/5 border border-white/10 rounded-3xl p-6 cursor-pointer text-center hover:bg-white/10 transition-all`;
          div.textContent = item.e;
          div.onclick = () => {
            if (item.n === correct.n) { score++; playSuccessSound(); next(); }
            else { playErrorSound(); }
            document.getElementById('fscore').textContent = `${score} / 5`;
          };
          optsDiv.appendChild(div);
        });
      }
      next();
    }

    function initClassifyGame(content) {
      const items = [{e:'🍎', type:'fruta'}, {e:'🥕', type:'verdura'}, {e:'🍌', type:'fruta'}, {e:'🥦', type:'verdura'}];
      let score = 0, round = 0, currentItem = null;
      content.innerHTML = `<h2 class="text-2xl font-bold text-center mb-6 uppercase tracking-widest text-blue-400">Sorting Unit</h2><div id="item-show" class="text-9xl mx-auto my-12 text-center"></div><div class="grid grid-cols-2 gap-8"><button onclick="classifyAnswer('fruta')" class="h-20 bg-blue-500/20 border border-blue-500/50 rounded-2xl text-blue-400 text-xl font-bold shadow-lg">FRUTA</button><button onclick="classifyAnswer('verdura')" class="h-20 bg-emerald-500/20 border border-emerald-500/50 rounded-2xl text-emerald-400 text-xl font-bold shadow-lg">VERDURA</button></div>`;
      introAudioClassify.play();
      window.classifyAnswer = (answer) => {
        if (answer === currentItem.type) { score++; playSuccessSound(); } else { playErrorSound(); }
        round++; if (round >= 4) showWinScreen(5); else nextClassify();
      };
      const nextClassify = () => {
        currentItem = items[Math.floor(Math.random()*items.length)];

    function initPuzzleGame(content) {
      content.innerHTML = `<h2 class="text-2xl font-bold text-center mb-4 uppercase tracking-widest text-blue-400">Reconstrucción</h2><div id="puzzle-grid" class="grid grid-cols-3 gap-3 mx-auto max-w-xs"></div>`;
      let puzzle = ['🍎','🍌','🍓','🍊','🥕','🥦','🌽','🍅',' '].sort(() => Math.random() - 0.5);
      introAudioPuzzle.play();
      const renderPuzzle = () => {
        const grid = document.getElementById('puzzle-grid'); grid.innerHTML = '';
        puzzle.forEach((tile, i) => {
          const div = document.createElement('div');
          div.className = `w-20 h-20 flex items-center justify-center text-4xl rounded-2xl border border-white/10 cursor-pointer ${tile === ' ' ? 'bg-black/40' : 'bg-white/5 hover:bg-white/10'}`;
          div.textContent = tile;
          div.onclick = () => {
            let emptyIdx = puzzle.indexOf(' ');
            if (Math.abs(Math.floor(i/3)-Math.floor(emptyIdx/3)) + Math.abs(i%3-emptyIdx%3) === 1) {
              [puzzle[i], puzzle[emptyIdx]] = [puzzle[emptyIdx], puzzle[i]];
              renderPuzzle();
              if (JSON.stringify(puzzle) === JSON.stringify(['🍎','🍌','🍓','🍊','🥕','🥦','🌽','🍅',' '])) showWinScreen(6);
            }
          };
          grid.appendChild(div);
        });
      }
      renderPuzzle();
    }

    function initShapeMatchGame(content) {
      let shapes = ['⭕','🔺','⬛','⭐'];
      let left = [...shapes], right = [...shapes].sort(() => Math.random()-0.5);
      let selected = null;
      content.innerHTML = `<h2 class="text-2xl font-bold text-center mb-6 uppercase tracking-widest text-blue-400">Core Link</h2><div class="flex justify-center gap-12"><div id="l" class="space-y-4"></div><div id="r" class="space-y-4"></div></div>`;
      introAudioShapes.play();
      const render = () => {
        document.getElementById('l').innerHTML = left.map((s,i) => `<div onclick="window.setShape(${i})" class="text-6xl p-4 bg-white/5 border border-white/10 rounded-2xl cursor-pointer hover:border-blue-500 transition-all">${s}</div>`).join('');
        document.getElementById('r').innerHTML = right.map((s,i) => `<div onclick="window.checkShape(${i})" class="text-6xl p-4 bg-white/5 border border-white/10 rounded-2xl cursor-pointer hover:border-emerald-500 transition-all">${s}</div>`).join('');
      }
      window.setShape = (i) => selected = i;
      window.checkShape = (i) => {
        if (selected !== null && left[selected] === right[i]) {
          playSuccessSound(); left.splice(selected, 1); right.splice(i, 1); selected = null;
          if (left.length === 0) showWinScreen(8); else render();
        } else playErrorSound();
      };
      render();
    }
